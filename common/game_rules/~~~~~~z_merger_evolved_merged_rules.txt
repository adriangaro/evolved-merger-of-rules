species_can_live_on_planet = {
	exists = root
	NAND = {
		root = {
			OR = {
				is_planet_class = pc_machine
				# Origin - F.C.S.S & M.C.S.S
				is_planet_class = pc_awaken_ai
			}
		}
		NOR = {
			has_trait = trait_machine_unit
			has_trait = trait_mechanical
			AND = {
				OR = {
					has_fcss = yes
					has_stellaris_evolved = yes
				}
				tec_is_cyborg_pop = yes
			}
			AND = {
				tec_is_cyborg_pop = yes
				exists = root.owner
				root.owner = {
					is_machine_empire = yes
					has_valid_civic = civic_machine_assimilator					# Assimilator empire
				}
			}
			# Girls Frontline
			has_trait = trait_doll
		}
	}
	# Expanded Gestalts
	NAND = {
		root = {
			NOT = {
				is_planet_class = pc_matryoshka
			}
		}
		has_trait = trait_machine_digital
	}
	# Human Fallen Empire
	NAND = {
		exists = root.controller
		root.controller = {
			is_country_type = hfe_plague_zombie
		}
	}
	# Vanilla
	if = {
		limit = {
			root = {
				is_planet_class = pc_hive
			}
		}
		OR = {
			has_trait = trait_hive_mind
			has_trait = trait_bugged_happy_hive			# Ethics & Civics Bug Branch
			AND = {
				#Necrophage Hive Mind can have Necrophytes
				exists = root.owner
				root.owner = {
					is_hive_empire = yes
					has_origin = origin_necrophage
					prev = {
						species_can_be_necrophaged = yes
					}
				}
			}
		}
	}
}

# Checks if a pop can be assembled
# This = Species
# Root = Planet
can_species_be_assembled = {
	is_sapient = yes
	OR = {
		has_stellaris_evolved = no
		tec_can_assemble_species_due_to_purity_assembly_policy = yes
	}
	NOT = {
		has_trait = trait_holodrone
	}
	# Holofrixit Revisited
	OR = {
		# Robots
		AND = {
			is_robotic = yes
			# Azur lane Stellaris DLC
			NOT = {
				has_trait = trait_robot_mind_relic_core
			}
			# Diagraphers Trait Mod
			NAND = {
				has_trait = trait_robot_rare_crystal_components
				exists = root.owner
				root.owner = {
					resource_stockpile_compare = {
						resource = rare_crystals
						value <= 0
					}
				}
			}
			# Vanilla
			root = {
				check_modifier_value = {
					modifier = planet_pop_assembly_add
					value > 0
				}
				owner = {
					NOR = {
						# Azur lane Stellaris DLC
						has_authority = auth_azur_lane_1
						has_authority = auth_azur_lane_2
						has_authority = auth_azur_lane_K
					}
				}
				# Stellaris Evolved
				OR = {
					has_stellaris_evolved = no
					if = {
						limit = {
							prev = {
								tec_is_self_replicating_pop = yes
							}
						}
						any_owned_species = {
							is_exact_same_species = prevprev
						}
					}
				}
				
				
			}
		}
		# Organics - this is the vanilla version
		AND = {
			# Evolved - Regular Organic rule - As closer as vanilla as possible
			is_organic_species = yes
			if = {
				limit = {
					OR = {
						has_trait = trait_plantoid_budding
						# Stellaris evolved
						tec_is_self_replicating_pop = yes
					}
				}
				root = {
					any_owned_species = {
						is_exact_same_species = prevprev
					}
					OR = {
						has_stellaris_evolved = no
						tec_blocks_organic_assembly = no
					}
				}
			}
			else_if = {
				# Hives can only assemble drones with spawning pools/queens only breed their own types
				limit = {
					root.owner = {
						is_hive_empire = yes
					}
				}
				AND = {
					OR = {
						has_trait = trait_hive_mind
						# Ethics and Civics - Bug Branch
						has_trait = trait_bugged_happy_hive
					}
					# Expanded Gestalts
					NOT = {
						has_trait = trait_hive_mind_overlord
					}
				}
				# Stellaris Evolved
				root = { 
					OR = {
						has_stellaris_evolved = no
						tec_can_assemble_organic_hive_species = yes
					}
				}	
			}can_support_branch_offices
			else_if = {
				limit = {
					root = { has_modifier = tiyanki_trophy }
				}
			}
			# Expanded Traits, Civics, Pops, and More
			else_if = {
				limit = {
					Root = {
						owner = {
							has_origin = origin_sacred_tree
						}
						OR = {
							has_deposit = d_tree_of_life_home
							has_deposit = is_for_colonizable
						}
					}
				}
				OR = {
					has_trait = trait_tree_caretaker
					merg_has_cloning_building = yes
				}
			}
			# Lithoid supremacy
			else_if = {
				limit = { has_trait = trait_pearl_born_royal }
				root = {
					any_owned_species = { 
						OR = {
							has_trait = trait_pearl_born_drone

						}
					}
				}
			}
